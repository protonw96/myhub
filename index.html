import pygame import random import asyncio import os # Инициализация pygame.init() WIDTH, HEIGHT = 400, 600 screen = pygame.display.set_mode((WIDTH, HEIGHT)) pygame.display.set_caption("Space Shooter: Menu & Shop") # Цвета WHITE = (255, 255, 255) BLACK = (0, 0, 0) YELLOW = (255, 255, 0) RED = (255, 0, 0) GREEN = (0, 255, 0) GRAY = (50, 50, 50) # Глобальные переменные прогресса coins = 0 shoot_delay = 500 max_hp = 3 class Particle: def __init__(self, x, y): self.x, self.y = x, y self.vx, self.vy = random.uniform(-3, 3), random.uniform(-3, 3) self.lifetime = 20 self.color = random.choice([YELLOW, RED, WHITE]) def update(self): self.x += self.vx self.y += self.vy self.lifetime -= 1 def draw(self, surface): if self.lifetime > 0: pygame.draw.circle(surface, self.color, (int(self.x), int(self.y)), 2) def load_asset(path): try: return pygame.image.load(path).convert_alpha() except: surf = pygame.Surface((30, 30)) surf.fill(RED) return surf async def main(): global coins, shoot_delay, max_hp font = pygame.font.SysFont("Arial", 24) big_font = pygame.font.SysFont("Arial", 40, bold=True) player_img = load_asset("assets/img/player.png") enemy_img = load_asset("assets/img/meteor.png") bullet_img = load_asset("assets/img/laser.png") state = "MENU" # Возможные состояния: MENU, GAME, SHOP while True: if state == "MENU": screen.fill(BLACK) title = big_font.render("SPACE HUB", True, WHITE) btn_play = font.render("1. ИГРАТЬ", True, GREEN) btn_shop = font.render("2. МАГАЗИН", True, YELLOW) coin_txt = font.render(f"Монеты: {coins}", True, WHITE) screen.blit(title, (WIDTH//2 - 90, 100)) screen.blit(btn_play, (WIDTH//2 - 50, 250)) screen.blit(btn_shop, (WIDTH//2 - 60, 310)) screen.blit(coin_txt, (WIDTH//2 - 50, 450)) for event in pygame.event.get(): if event.type == pygame.QUIT: return if event.type == pygame.KEYDOWN: if event.key == pygame.K_1: state = "GAME" if event.key == pygame.K_2: state = "SHOP" elif state == "SHOP": screen.fill(GRAY) title = font.render("МАГАЗИН УЛУЧШЕНИЙ", True, WHITE) info = font.render(f"Твои монеты: {coins}", True, YELLOW) item1 = font.render(f"1. Скорость атаки (50 монет)", True, WHITE) item2 = font.render(f"2. Доп. жизнь (30 монет)", True, WHITE) exit_txt = font.render("Нажми ESC для выхода в меню", True, BLACK) screen.blit(title, (50, 50)) screen.blit(info, (50, 100)) screen.blit(item1, (30, 200)) screen.blit(item2, (30, 260)) screen.blit(exit_txt, (50, 500)) for event in pygame.event.get(): if event.type == pygame.QUIT: return if event.type == pygame.KEYDOWN: if event.key == pygame.K_ESCAPE: state = "MENU" if event.key == pygame.K_1 and coins >= 50: shoot_delay = max(150, shoot_delay - 50) coins -= 50 if event.key == pygame.K_2 and coins >= 30: max_hp += 1 coins -= 30 elif state == "GAME": # Сброс параметров уровня player_rect = player_img.get_rect(midbottom=(WIDTH // 2, HEIGHT - 20)) bullets, enemies, particles = [], [], [] hp = max_hp score = 0 game_running = True last_shot = pygame.time.get_ticks() while game_running: current_time = pygame.time.get_ticks() screen.fill(BLACK) for event in pygame.event.get(): if event.type == pygame.QUIT: return if event.type == pygame.KEYDOWN: if event.key == pygame.K_ESCAPE: game_running = False keys = pygame.key.get_pressed() if keys[pygame.K_LEFT] and player_rect.left > 0: player_rect.x -= 5 if keys[pygame.K_RIGHT] and player_rect.right < WIDTH: player_rect.x += 5 if current_time - last_shot > shoot_delay: bullets.append(pygame.Rect(player_rect.centerx - 2, player_rect.top, 5, 15)) last_shot = current_time for b in bullets[:]: b.y -= 7 if b.bottom < 0: bullets.remove(b) if random.random() < 0.03: enemies.append(pygame.Rect(random.randint(0, WIDTH-30), -50, 30, 30)) for e in enemies[:]: e.y += 3 if e.top > HEIGHT: enemies.remove(e) hp -= 1 for b in bullets[:]: if e.colliderect(b): for _ in range(8): particles.append(Particle(e.centerx, e.centery)) if b in bullets: bullets.remove(b) if e in enemies: enemies.remove(e) score += 10 coins += 2 # Даем больше монет за сбитый метеорит break for p in particles[:]: p.update() if p.lifetime <= 0: particles.remove(p) # Отрисовка игры screen.blit(player_img, player_rect) for b in bullets: screen.blit(bullet_img, b) for e in enemies: screen.blit(enemy_img, e) for p in particles: p.draw(screen) txt = font.render(f"Очки: {score} HP: {hp} Монеты: {coins}", True, WHITE) screen.blit(txt, (10, 10)) if hp <= 0: game_running = False state = "MENU" pygame.display.flip() await asyncio.sleep(0) pygame.display.flip() await asyncio.sleep(0) asyncio.run(main())
