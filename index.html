рimport pygame import random import os import sys # --- ИНИЦИАЛИЗАЦИЯ И НАСТРОЙКИ ПУТЕЙ --- pygame.init() pygame.mixer.init() # Указываем пути к папкам BASE_DIR = os.path.dirname(__file__) IMG_DIR = os.path.join(BASE_DIR, 'assets', 'img') SND_DIR = os.path.join(BASE_DIR, 'assets', 'snd') # Константы WIDTH, HEIGHT = 480, 600 FPS = 60 WHITE = (255, 255, 255) BLACK = (0, 0, 0) METEOR_COUNT = 8 # Настройка окна screen = pygame.display.set_mode((WIDTH, HEIGHT)) pygame.display.set_caption("Space Battle: Final Edition") clock = pygame.time.Clock() # --- ФУНКЦИЯ БЕЗОПАСНОЙ ЗАГРУЗКИ --- def load_resource(type, folder, name): path = os.path.join(folder, name) try: if type == 'img': return pygame.image.load(path).convert_alpha() elif type == 'bg': return pygame.image.load(path).convert() elif type == 'snd': return pygame.mixer.Sound(path) except Exception: print(f"ОШИБКА: Файл не найден по пути: {path}") # Если файла нет, создаем "заглушку", чтобы игра не упала surf = pygame.Surface((50, 50)) surf.fill((255, 0, 0)) return surf if type != 'snd' else None # Загрузка ассетов BG_IMG = load_resource('bg', IMG_DIR, 'starfield.png') PLAYER_IMG = load_resource('img', IMG_DIR, 'player.png') METEOR_IMG = load_resource('img', IMG_DIR, 'meteor.png') LASER_IMG = load_resource('img', IMG_DIR, 'laser.png') SHOOT_SND = load_resource('snd', SND_DIR, 'pew.wav') EXPL_SND = load_resource('snd', SND_DIR, 'expl.wav') # Загрузка музыки (отдельно) try: pygame.mixer.music.load(os.path.join(SND_DIR, 'music.ogg')) pygame.mixer.music.set_volume(0.2) except: print("Музыка не найдена, играем в тишине") # --- ВСПОМОГАТЕЛЬНЫЕ ФУНКЦИИ --- def draw_text(text, size, x, y): font = pygame.font.SysFont("arial", size, bold=True) surface = font.render(text, True, WHITE) rect = surface.get_rect(midtop=(x, y)) screen.blit(surface, rect) # --- КЛАССЫ --- class Player(pygame.sprite.Sprite): def __init__(self): super().__init__() self.image = pygame.transform.scale(PLAYER_IMG, (50, 38)) self.rect = self.image.get_rect(centerx=WIDTH//2, bottom=HEIGHT-10) self.radius = 20 self.last_shot = pygame.time.get_ticks() self.delay = 250 def update(self): keys = pygame.key.get_pressed() if keys[pygame.K_LEFT] or keys[pygame.K_a]: self.rect.x -= 8 if keys[pygame.K_RIGHT] or keys[pygame.K_d]: self.rect.x += 8 self.rect.clamp_ip(screen.get_rect()) # Не дает выйти за экран def shoot(self, sprites, bullets): now = pygame.time.get_ticks() if now - self.last_shot > self.delay: self.last_shot = now b = Bullet(self.rect.centerx, self.rect.top) sprites.add(b) bullets.add(b) if SHOOT_SND: SHOOT_SND.play() class Meteor(pygame.sprite.Sprite): def __init__(self): super().__init__() size = random.randint(30, 70) self.image_orig = pygame.transform.scale(METEOR_IMG, (size, size)) self.image = self.image_orig.copy() self.rect = self.image.get_rect() self.radius = int(self.rect.width * 0.8 / 2) self.spawn() self.rot = 0 self.rot_speed = random.randint(-5, 5) def spawn(self): self.rect.x = random.randrange(WIDTH - self.rect.width) self.rect.y = random.randrange(-100, -40) self.speedy = random.randint(2, 6) self.speedx = random.randint(-1, 1) def update(self): self.rot = (self.rot + self.rot_speed) % 360 self.image = pygame.transform.rotate(self.image_orig, self.rot) self.rect = self.image.get_rect(center=self.rect.center) self.rect.y += self.speedy self.rect.x += self.speedx if self.rect.top > HEIGHT: self.spawn() class Bullet(pygame.sprite.Sprite): def __init__(self, x, y): super().__init__() self.image = LASER_IMG self.rect = self.image.get_rect(midbottom=(x, y)) def update(self): self.rect.y -= 10 if self.rect.bottom < 0: self.kill() # --- ЭКРАНЫ --- def show_screen(title, subtitle, score=None): screen.blit(BG_IMG, (0, 0)) draw_text(title, 64, WIDTH // 2, HEIGHT / 4) if score is not None: draw_text(f"SCORE: {score}", 32, WIDTH // 2, HEIGHT / 2) draw_text(subtitle, 24, WIDTH // 2, HEIGHT * 0.75) pygame.display.flip() waiting = True while waiting: clock.tick(FPS) for event in pygame.event.get(): if event.type == pygame.QUIT: pygame.quit(); sys.exit() if event.type == pygame.KEYDOWN: waiting = False # --- ГЛАВНЫЙ ЦИКЛ --- def game_loop(): while True: # 1. Меню show_screen("SPACE BATTLE", "Press any key to START") # 2. Инициализация уровня all_sprites = pygame.sprite.Group() mobs = pygame.sprite.Group() bullets = pygame.sprite.Group() player = Player() all_sprites.add(player) for _ in range(METEOR_COUNT): m = Meteor() all_sprites.add(m) mobs.add(m) score = 0 if pygame.mixer.music.get_busy() == False: try: pygame.mixer.music.play(-1) except: pass # 3. Геймплей running = True while running: clock.tick(FPS) for event in pygame.event.get(): if event.type == pygame.QUIT: pygame.quit(); sys.exit() player.shoot(all_sprites, bullets) all_sprites.update() # Коллизии hits = pygame.sprite.groupcollide(mobs, bullets, True, True, pygame.sprite.collide_circle) for hit in hits: score += 10 if EXPL_SND: EXPL_SND.play() m = Meteor() all_sprites.add(m) mobs.add(m) if pygame.sprite.spritecollide(player, mobs, False, pygame.sprite.collide_circle): pygame.mixer.music.stop() running = False # Переход к Game Over # Отрисовка screen.blit(BG_IMG, (0, 0)) all_sprites.draw(screen) draw_text(f"Score: {score}", 20, WIDTH // 2, 10) pygame.display.flip() # 4. Экран конца игры show_screen("GAME OVER", "Press any key to RESTART", score) if __name__ == "__main__": game_loop()
