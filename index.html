import pygame import random import os import sys # --- НАСТРОЙКИ ПУТЕЙ --- BASE_DIR = os.path.dirname(__file__) IMG_DIR = os.path.join(BASE_DIR, 'assets', 'img') SND_DIR = os.path.join(BASE_DIR, 'assets', 'snd') # --- КОНСТАНТЫ --- WIDTH, HEIGHT = 480, 600 FPS = 60 WHITE = (255, 255, 255) BLACK = (0, 0, 0) METEOR_COUNT = 8 # --- ИНИЦИАЛИЗАЦИЯ --- pygame.init() pygame.mixer.init() screen = pygame.display.set_mode((WIDTH, HEIGHT)) pygame.display.set_caption("Space Battle: Final") clock = pygame.time.Clock() # --- ФУНКЦИЯ БЕЗОПАСНОЙ ЗАГРУЗКИ --- def load_img(name, use_alpha=True): path = os.path.join(IMG_DIR, name) try: img = pygame.image.load(path) return img.convert_alpha() if use_alpha else img.convert() except: print(f"ПРЕДУПРЕЖДЕНИЕ: Файл {name} не найден. Создана заглушка.") surf = pygame.Surface((50, 50)) surf.fill((random.randint(0,255), 0, 0)) return surf def load_snd(name): path = os.path.join(SND_DIR, name) try: return pygame.mixer.Sound(path) except: return None # --- ЗАГРУЗКА РЕСУРСОВ --- BG_IMG = load_img('starfield.png', False) # Масштабируем фон под размер окна BG_IMG = pygame.transform.scale(BG_IMG, (WIDTH, HEIGHT)) PLAYER_IMG = load_img('player.png') METEOR_IMG = load_img('meteor.png') LASER_IMG = load_img('laser.png') SHOOT_SND = load_snd('pew.wav') EXPL_SND = load_snd('expl.wav') try: pygame.mixer.music.load(os.path.join(SND_DIR, 'music.ogg')) pygame.mixer.music.set_volume(0.2) except: pass # --- КЛАССЫ --- def draw_text(surf, text, size, x, y): font = pygame.font.SysFont("arial", size, bold=True) text_surface = font.render(text, True, WHITE) text_rect = text_surface.get_rect(midtop=(x, y)) surf.blit(text_surface, text_rect) class Player(pygame.sprite.Sprite): def __init__(self): super().__init__() self.image = pygame.transform.scale(PLAYER_IMG, (50, 38)) self.rect = self.image.get_rect(centerx=WIDTH // 2, bottom=HEIGHT - 10) self.radius = 20 self.last_shot = pygame.time.get_ticks() def update(self): keys = pygame.key.get_pressed() if keys[pygame.K_LEFT] or keys[pygame.K_a]: self.rect.x -= 8 if keys[pygame.K_RIGHT] or keys[pygame.K_d]: self.rect.x += 8 if self.rect.right > WIDTH: self.rect.right = WIDTH if self.rect.left < 0: self.rect.left = 0 def shoot(self, all_sprites, bullets): now = pygame.time.get_ticks() if now - self.last_shot > 250: self.last_shot = now b = Bullet(self.rect.centerx, self.rect.top) all_sprites.add(b) bullets.add(b) if SHOOT_SND: SHOOT_SND.play() class Meteor(pygame.sprite.Sprite): def __init__(self): super().__init__() size = random.randint(30, 60) self.image_orig = pygame.transform.scale(METEOR_IMG, (size, size)) self.image = self.image_orig.copy() self.rect = self.image.get_rect() self.radius = int(self.rect.width * 0.8 / 2) self.spawn() self.rot = 0 self.rot_speed = random.randint(-5, 5) def spawn(self): self.rect.x = random.randrange(WIDTH - self.rect.width) self.rect.y = random.randrange(-100, -40) self.speedy = random.randint(2, 7) self.speedx = random.randint(-1, 1) def update(self): self.rot = (self.rot + self.rot_speed) % 360 self.image = pygame.transform.rotate(self.image_orig, self.rot) self.rect = self.image.get_rect(center=self.rect.center) self.rect.y += self.speedy self.rect.x += self.speedx if self.rect.top > HEIGHT: self.spawn() class Bullet(pygame.sprite.Sprite): def __init__(self, x, y): super().__init__() self.image = pygame.transform.scale(LASER_IMG, (10, 20)) self.rect = self.image.get_rect(midbottom=(x, y)) def update(self): self.rect.y -= 10 if self.rect.bottom < 0: self.kill() # --- ЭКРАНЫ --- def show_screen(title, subtitle, score=None): screen.blit(BG_IMG, (0, 0)) draw_text(screen, title, 64, WIDTH // 2, HEIGHT / 4) if score is not None: draw_text(screen, f"FINAL SCORE: {score}", 36, WIDTH // 2, HEIGHT / 2) draw_text(screen, subtitle, 22, WIDTH // 2, HEIGHT * 0.75) pygame.display.flip() waiting = True while waiting: clock.tick(FPS) for event in pygame.event.get(): if event.type == pygame.QUIT: pygame.quit(); sys.exit() if event.type == pygame.KEYDOWN or event.type == pygame.MOUSEBUTTONDOWN: waiting = False # --- ГЛАВНЫЙ ЦИКЛ --- def main(): while True: show_screen("SPACE BATTLE", "Press any key to START") all_sprites = pygame.sprite.Group() mobs = pygame.sprite.Group() bullets = pygame.sprite.Group() player = Player() all_sprites.add(player) for _ in range(METEOR_COUNT): m = Meteor() all_sprites.add(m) mobs.add(m) score = 0 try: pygame.mixer.music.play(-1) except: pass running = True while running: clock.tick(FPS) for event in pygame.event.get(): if event.type == pygame.QUIT: pygame.quit(); sys.exit() # Обновление player.shoot(all_sprites, bullets) all_sprites.update() # Коллизии hits = pygame.sprite.groupcollide(mobs, bullets, True, True, pygame.sprite.collide_circle) for hit in hits: score += 10 if EXPL_SND: EXPL_SND.play() m = Meteor() all_sprites.add(m) mobs.add(m) if pygame.sprite.spritecollide(player, mobs, False, pygame.sprite.collide_circle): pygame.mixer.music.stop() running = False # Отрисовка screen.fill(BLACK) screen.blit(BG_IMG, (0, 0)) all_sprites.draw(screen) draw_text(screen, f"Score: {score}", 20, WIDTH // 2, 10) pygame.display.flip() show_screen("GAME OVER", "Press any key to RESTART", score) if __name__ == "__main__": main()
