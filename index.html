import pygame import random import os import sys # --- НАСТРОЙКИ --- WIDTH, HEIGHT = 480, 600 FPS = 60 WHITE = (255, 255, 255) BLACK = (0, 0, 0) GOLD = (255, 215, 0) # Пути BASE_DIR = os.path.dirname(__file__) IMG_DIR = os.path.join(BASE_DIR, 'assets', 'img') SND_DIR = os.path.join(BASE_DIR, 'assets', 'snd') pygame.init() pygame.mixer.init() screen = pygame.display.set_mode((WIDTH, HEIGHT)) pygame.display.set_caption("Space Battle: Levels 1-5") clock = pygame.time.Clock() # --- ЗАГРУЗКА --- def load_img(name, alpha=True): path = os.path.join(IMG_DIR, name) try: img = pygame.image.load(path) return img.convert_alpha() if alpha else img.convert() except: s = pygame.Surface((40, 40)); s.fill((200, 0, 0)) return s BG_IMG = pygame.transform.scale(load_img('starfield.png', False), (WIDTH, HEIGHT)) PLAYER_IMG = load_img('player.png') METEOR_IMG = load_img('meteor.png') LASER_IMG = load_img('laser.png') SHOOT_SND = os.path.join(SND_DIR, 'pew.wav') # Загрузим позже в классе EXPL_SND = os.path.join(SND_DIR, 'expl.wav') # --- КЛАССЫ --- def draw_text(surf, text, size, x, y, color=WHITE): font = pygame.font.SysFont("arial", size, bold=True) txt = font.render(text, True, color) rect = txt.get_rect(midtop=(x, y)) surf.blit(txt, rect) class Player(pygame.sprite.Sprite): def __init__(self): super().__init__() self.image = pygame.transform.scale(PLAYER_IMG, (50, 38)) self.rect = self.image.get_rect(centerx=WIDTH//2, bottom=HEIGHT-10) self.shoot_delay = 300 self.last_shot = pygame.time.get_ticks() def update(self): keys = pygame.key.get_pressed() if keys[pygame.K_LEFT] or keys[pygame.K_a]: self.rect.x -= 8 if keys[pygame.K_RIGHT] or keys[pygame.K_d]: self.rect.x += 8 self.rect.clamp_ip(screen.get_rect()) def shoot(self, sprites, bullets, level): now = pygame.time.get_ticks() # С каждым уровнем стреляем чуть быстрее current_delay = self.shoot_delay - (level * 20) if now - self.last_shot > current_delay: self.last_shot = now b = Bullet(self.rect.centerx, self.rect.top) sprites.add(b) bullets.add(b) try: pygame.mixer.Sound(SHOOT_SND).play() except: pass class Meteor(pygame.sprite.Sprite): def __init__(self, speed_mod): super().__init__() size = random.randint(30, 60) self.image_orig = pygame.transform.scale(METEOR_IMG, (size, size)) self.image = self.image_orig.copy() self.rect = self.image.get_rect() self.radius = int(self.rect.width * 0.8 / 2) self.speed_mod = speed_mod # Влияние уровня на скорость self.spawn() def spawn(self): self.rect.x = random.randrange(WIDTH - self.rect.width) self.rect.y = random.randrange(-150, -50) self.speedy = random.randint(2, 5) + self.speed_mod self.speedx = random.randint(-1, 1) def update(self): self.rect.y += self.speedy self.rect.x += self.speedx if self.rect.top > HEIGHT: self.spawn() class Bullet(pygame.sprite.Sprite): def __init__(self, x, y): super().__init__() self.image = pygame.transform.scale(LASER_IMG, (10, 20)) self.rect = self.image.get_rect(midbottom=(x, y)) def update(self): self.rect.y -= 12 if self.rect.bottom < 0: self.kill() # --- ГЛАВНАЯ ФУНКЦИЯ --- def main(): while True: # Экран старта screen.blit(BG_IMG, (0, 0)) draw_text(screen, "SPACE BATTLE", 60, WIDTH//2, HEIGHT/3) draw_text(screen, "LEVELS 1 - 5", 30, WIDTH//2, HEIGHT/2, GOLD) draw_text(screen, "Press any key", 20, WIDTH//2, HEIGHT*0.8) pygame.display.flip() wait = True while wait: for e in pygame.event.get(): if e.type == pygame.QUIT: pygame.quit(); sys.exit() if e.type == pygame.KEYDOWN: wait = False # Старт игры all_sprites = pygame.sprite.Group() mobs = pygame.sprite.Group() bullets = pygame.sprite.Group() player = Player() all_sprites.add(player) score = 0 level = 1 meteor_count = 6 # Спавн первых метеоритов for _ in range(meteor_count): m = Meteor(level) all_sprites.add(m) mobs.add(m) running = True while running: clock.tick(FPS) for e in pygame.event.get(): if e.type == pygame.QUIT: pygame.quit(); sys.exit() # ЛОГИКА УРОВНЕЙ # 0-500: Lvl 1, 500-1200: Lvl 2, и так далее new_level = 1 if score > 2000: new_level = 5 elif score > 1200: new_level = 4 elif score > 600: new_level = 3 elif score > 250: new_level = 2 if new_level > level: level = new_level # Добавляем еще один метеорит при повышении уровня m = Meteor(level) all_sprites.add(m) mobs.add(m) # Обновление player.shoot(all_sprites, bullets, level) all_sprites.update() # Коллизии hits = pygame.sprite.groupcollide(mobs, bullets, True, True) for h in hits: score += 10 try: pygame.mixer.Sound(EXPL_SND).play() except: pass m = Meteor(level) all_sprites.add(m) mobs.add(m) if pygame.sprite.spritecollide(player, mobs, False, pygame.sprite.collide_circle): running = False # Отрисовка screen.blit(BG_IMG, (0, 0)) all_sprites.draw(screen) draw_text(screen, f"SCORE: {score}", 20, 70, 10) draw_text(screen, f"LEVEL: {level}", 20, WIDTH - 70, 10, GOLD) if level == 5: draw_text(screen, "MAX LEVEL!", 15, WIDTH//2, 40, GOLD) pygame.display.flip() # GAME OVER screen.blit(BG_IMG, (0, 0)) draw_text(screen, "GAME OVER", 60, WIDTH//2, HEIGHT/3) draw_text(screen, f"FINAL SCORE: {score}", 30, WIDTH//2, HEIGHT/2) draw_text(screen, f"REACHED LEVEL: {level}", 25, WIDTH//2, HEIGHT/1.7, GOLD) draw_text(screen, "Press key to Restart", 20, WIDTH//2, HEIGHT*0.8) pygame.display.flip() pygame.time.wait(1000) wait = True while wait: for e in pygame.event.get(): if e.type == pygame.QUIT: pygame.quit(); sys.exit() if e.type == pygame.KEYDOWN: wait = False if __name__ == "__main__": main()
