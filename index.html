import pygame import random import sys # --- НАСТРОЙКИ --- WIDTH, HEIGHT = 480, 600 FPS = 60 # Цвета WHITE = (255, 255, 255) BLACK = (0, 0, 0) GOLD = (255, 215, 0) GRAY = (50, 50, 50) RED = (255, 50, 50) GREEN = (50, 255, 50) BLUE = (50, 50, 255) pygame.init() screen = pygame.display.set_mode((WIDTH, HEIGHT)) pygame.display.set_caption("Space Battle: Pro Edition") clock = pygame.time.Clock() # --- ДАННЫЕ ИГРОКА --- player_data = { "money": 100, "max_hp": 3, "shoot_speed": 400, # Задержка в мс (меньше = быстрее) "bullet_damage": 1, "move_speed": 5, "levels_unlocked": 1 } # --- КЛАССЫ ИГРЫ --- class Player(pygame.sprite.Sprite): def __init__(self): super().__init__() self.image = pygame.Surface((40, 40)) self.image.fill(GREEN) self.rect = self.image.get_rect(midbottom=(WIDTH//2, HEIGHT - 20)) self.hp = player_data["max_hp"] self.last_shot = pygame.time.get_ticks() def update(self): keys = pygame.key.get_pressed() if keys[pygame.K_LEFT] and self.rect.left > 0: self.rect.x -= player_data["move_speed"] if keys[pygame.K_RIGHT] and self.rect.right < WIDTH: self.rect.x += player_data["move_speed"] if keys[pygame.K_SPACE]: self.shoot() def shoot(self): now = pygame.time.get_ticks() if now - self.last_shot > player_data["shoot_speed"]: bullet = Bullet(self.rect.centerx, self.rect.top) all_sprites.add(bullet) bullets.add(bullet) self.last_shot = now class Enemy(pygame.sprite.Sprite): def __init__(self, level): super().__init__() self.image = pygame.Surface((30, 30)) # Чем выше уровень, тем "краснее" враги color_val = min(255, 50 + level * 10) self.image.fill((color_val, 0, 0)) self.rect = self.image.get_rect(x=random.randint(0, WIDTH-30), y=random.randint(-100, -40)) # Характеристики врага растут от уровня self.speed = random.randint(2, 4) + (level // 3) self.hp = 1 + (level // 5) def update(self): self.rect.y += self.speed if self.rect.top > HEIGHT: self.rect.y = random.randint(-100, -40) self.rect.x = random.randint(0, WIDTH-30) class Bullet(pygame.sprite.Sprite): def __init__(self, x, y): super().__init__() self.image = pygame.Surface((5, 15)) self.image.fill(GOLD) self.rect = self.image.get_rect(center=(x, y)) def update(self): self.rect.y -= 10 if self.rect.bottom < 0: self.kill() # --- ВСПОМОГАТЕЛЬНЫЕ ФУНКЦИИ --- def draw_text(surf, text, size, x, y, color=WHITE): font = pygame.font.SysFont("arial", size, bold=True) txt = font.render(str(text), True, color) rect = txt.get_rect(midtop=(x, y)) surf.blit(txt, rect) return rect def draw_button(surf, text, x, y, w, h, color, hover_color): mouse = pygame.mouse.get_pos() click = pygame.mouse.get_pressed() rect = pygame.Rect(x, y, w, h) res = False current_color = hover_color if rect.collidepoint(mouse) else color pygame.draw.rect(surf, current_color, rect, border_radius=10) pygame.draw.rect(surf, WHITE, rect, 2, border_radius=10) draw_text(surf, text, 20, x + w//2, y + h//2 - 12) if rect.collidepoint(mouse) and click[0]: pygame.time.wait(150) res = True return res # --- ЭКРАНЫ --- def main_menu(): while True: screen.fill(BLACK) draw_text(screen, "SPACE BATTLE", 50, WIDTH//2, 80, GOLD) draw_text(screen, f"ДЕНЬГИ: ${player_data['money']}", 25, WIDTH//2, 150, GREEN) if draw_button(screen, "В БОЙ", WIDTH//2 - 100, 250, 200, 60, GRAY, GREEN): level_select() if draw_button(screen, "МАГАЗИН", WIDTH//2 - 100, 330, 200, 60, GRAY, GOLD): shop_menu() if draw_button(screen, "ВЫХОД", WIDTH//2 - 100, 410, 200, 60, GRAY, RED): pygame.quit(); sys.exit() pygame.display.flip() for e in pygame.event.get(): if e.type == pygame.QUIT: pygame.quit(); sys.exit() def shop_menu(): while True: screen.fill(BLACK) draw_text(screen, "УЛУЧШЕНИЯ", 40, WIDTH//2, 30, GOLD) draw_text(screen, f"БАЛАНС: ${player_data['money']}", 20, WIDTH//2, 80, GREEN) # Цены (можно усложнить формулой) costs = {"hp": 150, "dmg": 300, "speed": 200, "fire": 250} if draw_button(screen, f"+1 HP (${costs['hp']})", 90, 150, 300, 50, GRAY, BLUE): if player_data["money"] >= costs["hp"]: player_data["money"] -= costs["hp"] player_data["max_hp"] += 1 if draw_button(screen, f"+УРОН (${costs['dmg']})", 90, 210, 300, 50, GRAY, BLUE): if player_data["money"] >= costs["dmg"]: player_data["money"] -= costs["dmg"] player_data["bullet_damage"] += 1 if draw_button(screen, f"+СКОРОСТЬ ХОДА (${costs['speed']})", 90, 270, 300, 50, GRAY, BLUE): if player_data["money"] >= costs["speed"]: player_data["money"] -= costs["speed"] player_data["move_speed"] += 1 if draw_button(screen, f"+СКОРОСТРЕЛЬНОСТЬ (${costs['fire']})", 90, 330, 300, 50, GRAY, BLUE): if player_data["money"] >= costs["fire"]: player_data["money"] -= costs["fire"] player_data["shoot_speed"] = max(100, player_data["shoot_speed"] - 50) if draw_button(screen, "НАЗАД", WIDTH//2 - 80, 450, 160, 50, RED, WHITE): return pygame.display.flip() for e in pygame.event.get(): if e.type == pygame.QUIT: pygame.quit(); sys.exit() def level_select(): while True: screen.fill(BLACK) draw_text(screen, "ВЫБЕРИТЕ УРОВЕНЬ", 35, WIDTH//2, 40, GOLD) # Сетка уровней 3 ряда по 5 (итого 15 уровней) for i in range(1, 16): col = (i - 1) % 5 row = (i - 1) // 5 x = 50 + col * 80 y = 150 + row * 80 color = GREEN if i <= player_data["levels_unlocked"] else GRAY if draw_button(screen, str(i), x, y, 60, 60, color, GOLD): if i <= player_data["levels_unlocked"]: game_result = run_game(i) if game_result == "WIN": if i == player_data["levels_unlocked"]: player_data["levels_unlocked"] += 1 if draw_button(screen, "В МЕНЮ", WIDTH//2 - 80, 480, 160, 50, RED, WHITE): return pygame.display.flip() for e in pygame.event.get(): if e.type == pygame.QUIT: pygame.quit(); sys.exit() # --- ГЕЙМПЛЕЙ --- all_sprites = pygame.sprite.Group() enemies = pygame.sprite.Group() bullets = pygame.sprite.Group() def run_game(level): global all_sprites, enemies, bullets all_sprites = pygame.sprite.Group() enemies = pygame.sprite.Group() bullets = pygame.sprite.Group() player = Player() all_sprites.add(player) # Количество врагов зависит от уровня enemy_count = 5 + level for _ in range(enemy_count): e = Enemy(level) all_sprites.add(e) enemies.add(e) score = 0 required_kills = 5 + level * 2 running = True while running: clock.tick(FPS) screen.fill(BLACK) # Отрисовка интерфейса в бою draw_text(screen, f"УРОВЕНЬ: {level}", 20, 70, 10, GOLD) draw_text(screen, f"HP: {player.hp}", 20, 200, 10, RED) draw_text(screen, f"ЦЕЛЬ: {score}/{required_kills}", 20, 350, 10, WHITE) for event in pygame.event.get(): if event.type == pygame.QUIT: pygame.quit(); sys.exit() if event.type == pygame.KEYDOWN and event.key == pygame.K_ESCAPE: return "QUIT" all_sprites.update() # Столкновение пуль с врагами hits = pygame.sprite.group_collide(enemies, bullets, False, True) for hit in hits: hit.hp -= player_data["bullet_damage"] if hit.hp <= 0: score += 1 player_data["money"] += 10 # Деньги за убийство hit.kill() # Спавним нового врага, пока цель не достигнута if score + len(enemies) < required_kills: new_enemy = Enemy(level) all_sprites.add(new_enemy) enemies.add(new_enemy) # Столкновение врагов с игроком damage_hits = pygame.sprite.spritecollide(player, enemies, True) if damage_hits: player.hp -= 1 if player.hp <= 0: return "LOSE" # Восполняем врага new_enemy = Enemy(level) all_sprites.add(new_enemy) enemies.add(new_enemy) # Проверка победы if score >= required_kills: # Бонус за уровень player_data["money"] += level * 50 return "WIN" all_sprites.draw(screen) pygame.display.flip() if __name__ == "__main__": main_menu()
