import pygame import random import os import sys # --- НАСТРОЙКИ ПУТЕЙ --- game_dir = os.path.dirname(__file__) assets_dir = os.path.join(game_dir, 'assets') img_dir = os.path.join(assets_dir, 'img') snd_dir = os.path.join(assets_dir, 'snd') # --- КОНФИГУРАЦИЯ --- WIDTH = 480 HEIGHT = 600 FPS = 60 METEOR_COUNT = 8 WHITE = (255, 255, 255) BLACK = (0, 0, 0) # Центрирование окна os.environ['SDL_VIDEO_CENTERED'] = '1' # --- ВСПОМОГАТЕЛЬНЫЕ ФУНКЦИИ --- def draw_text(surf, text, size, x, y): """Функция для отрисовки текста на экране.""" font = pygame.font.SysFont("arial", size, bold=True) text_surface = font.render(text, True, WHITE) text_rect = text_surface.get_rect(midtop=(x, y)) surf.blit(text_surface, text_rect) def show_menu(screen, bg_img): """Экран начального меню.""" screen.blit(bg_img, (0, 0)) draw_text(screen, "SPACE BATTLE", 48, WIDTH // 2, HEIGHT / 4) draw_text(screen, "Arrows or AD to move", 22, WIDTH // 2, HEIGHT / 2) draw_text(screen, "Press any key to Start", 26, WIDTH // 2, HEIGHT * 0.7) pygame.display.flip() waiting = True while waiting: for event in pygame.event.get(): if event.type == pygame.QUIT: pygame.quit() sys.exit() if event.type == pygame.KEYDOWN: waiting = False def show_go_screen(screen, bg_img, score): """Экран Game Over с результатом.""" screen.blit(bg_img, (0, 0)) draw_text(screen, "GAME OVER", 64, WIDTH // 2, HEIGHT / 4) draw_text(screen, f"FINAL SCORE: {score}", 36, WIDTH // 2, HEIGHT / 2) draw_text(screen, "Press any key to restart", 22, WIDTH // 2, HEIGHT * 0.75) pygame.display.flip() pygame.time.wait(500) # Короткая пауза, чтобы случайно не пропустить экран waiting = True while waiting: for event in pygame.event.get(): if event.type == pygame.QUIT: pygame.quit() sys.exit() if event.type == pygame.KEYDOWN: waiting = False # --- КЛАССЫ ОБЪЕКТОВ --- class Player(pygame.sprite.Sprite): def __init__(self, img): super().__init__() self.image = pygame.transform.scale(img, (50, 38)) self.rect = self.image.get_rect(centerx=WIDTH // 2, bottom=HEIGHT - 10) self.radius = 20 self.shoot_delay = 250 self.last_shot = pygame.time.get_ticks() def update(self): keystate = pygame.key.get_pressed() if keystate[pygame.K_LEFT] or keystate[pygame.K_a]: self.rect.x -= 8 if keystate[pygame.K_RIGHT] or keystate[pygame.K_d]: self.rect.x += 8 if self.rect.right > WIDTH: self.rect.right = WIDTH if self.rect.left < 0: self.rect.left = 0 def shoot(self, bullet_img, snd, all_sprites, bullet_group): now = pygame.time.get_ticks() if now - self.last_shot > self.shoot_delay: self.last_shot = now b = Bullet(self.rect.centerx, self.rect.top, bullet_img) all_sprites.add(b) bullet_group.add(b) if snd: snd.play() class Meteor(pygame.sprite.Sprite): def __init__(self, sizes): super().__init__() self.image_orig = random.choice(sizes) self.image = self.image_orig.copy() self.rect = self.image.get_rect() self.radius = int(self.rect.width * 0.85 / 2) self.spawn() self.rot = 0 self.rot_speed = random.randrange(-8, 8) self.last_update = pygame.time.get_ticks() def spawn(self): self.rect.x = random.randrange(WIDTH - self.rect.width) self.rect.y = random.randrange(-150, -100) self.speedy = random.randrange(1, 8) self.speedx = random.randrange(-2, 2) def update(self): # Вращение now = pygame.time.get_ticks() if now - self.last_update > 50: self.last_update = now self.rot = (self.rot + self.rot_speed) % 360 new_image = pygame.transform.rotate(self.image_orig, self.rot) old_center = self.rect.center self.image = new_image self.rect = self.image.get_rect(center=old_center) self.rect.y += self.speedy self.rect.x += self.speedx if self.rect.top > HEIGHT + 10: self.spawn() class Bullet(pygame.sprite.Sprite): def __init__(self, x, y, img): super().__init__() self.image = img self.rect = self.image.get_rect(midbottom=(x, y)) def update(self): self.rect.y -= 12 if self.rect.bottom < 0: self.kill() # --- ГЛАВНАЯ ФУНКЦИЯ --- def main(): pygame.init() pygame.mixer.init() screen = pygame.display.set_mode((WIDTH, HEIGHT)) pygame.display.set_caption("Space Battle Offline") clock = pygame.time.Clock() # Загрузка ресурсов try: bg = pygame.image.load(os.path.join(img_dir, "starfield.png")).convert() p_img = pygame.image.load(os.path.join(img_dir, "player.png")).convert_alpha() m_img = pygame.image.load(os.path.join(img_dir, "meteor.png")).convert_alpha() b_img = pygame.image.load(os.path.join(img_dir, "laser.png")).convert_alpha() m_sizes = [pygame.transform.scale(m_img, (s, s)) for s in [30, 45, 60]] s_shoot = pygame.mixer.Sound(os.path.join(snd_dir, "pew.wav")) s_expl = pygame.mixer.Sound(os.path.join(snd_dir, "expl.wav")) pygame.mixer.music.load(os.path.join(snd_dir, "music.ogg")) pygame.mixer.music.set_volume(0.2) except Exception as e: print(f"Ошибка: {e}. Проверь папку assets!") pygame.quit() sys.exit() while True: # Глобальный цикл для рестартов show_menu(screen, bg) pygame.mixer.music.play(-1) all_sprites = pygame.sprite.Group() mobs = pygame.sprite.Group() bullets = pygame.sprite.Group() player = Player(p_img) all_sprites.add(player) for i in range(METEOR_COUNT): m = Meteor(m_sizes) all_sprites.add(m) mobs.add(m) score = 0 game_running = True while game_running: clock.tick(FPS) for event in pygame.event.get(): if event.type == pygame.QUIT: pygame.quit() sys.exit() # Обновление player.shoot(b_img, s_shoot, all_sprites, bullets) all_sprites.update() # Коллизии Пуля - Метеор hits = pygame.sprite.groupcollide(mobs, bullets, True, True, pygame.sprite.collide_circle) for hit in hits: score += 10 s_expl.play() m = Meteor(m_sizes) all_sprites.add(m) mobs.add(m) # Коллизия Игрок - Метеор if pygame.sprite.spritecollide(player, mobs, False, pygame.sprite.collide_circle): pygame.mixer.music.stop() show_go_screen(screen, bg, score) game_running = False # Выход из игрового цикла в меню # Отрисовка screen.blit(bg, (0, 0)) all_sprites.draw(screen) draw_text(screen, f"Score: {score}", 20, WIDTH // 2, 10) pygame.display.flip() if __name__ == "__main__": main()
